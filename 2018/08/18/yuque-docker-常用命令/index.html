    • docker 常用命令

    1. 基础命令
    # 命令格式
docker 命令关键字(command) 一系列参数([args...])

# 查询命令帮助
docker command --help

    1. 启动容器
      docker start id/name

    2. 关闭容器
      docker stop id/name

    3. 查询在线容器

    docker ps [-a]

# -a 显示所有，包括已经关闭的

    1. 查询容器信息
      docker inspect id/name

    2. 查看容器日志
      docker logs [-f] id/name

    3. 查看容器资源占用
      docker stats

    4. 容器内部命令

    # 在容器内部执行命令
docker exec id/name 执行命令
# 以root用户执行命令
docker exec -it id/name 执行命令
# 以root用户打开一个终端,exit退出
docker exec -it id/name /bin/bash
    1. 镜像查询
    docker search xx
    1. 镜像下载
    docker pull xx
    1. 本地镜像列表
    docker images
    1. 启动容器
    docker run [--name] [--link] 镜像id/name

# --link 容器连接
# --name 容器起的名称
# --p 端口映射 8080:80
# --e 环境变量设置
    1. 删除容器
    docker rm id/name
    1. 创建镜像
    docker commit id imageName
    1. 关闭所有运行的容器
    docker stop $(docker ps -q)
